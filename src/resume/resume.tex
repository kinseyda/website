	\noindent
	\def\@tempa{#2}
	\begin{large}
		\textbf{#1}\ifx\@tempa\@empty\relax \else\ --- #2\fi \par
	\end{large}
	\def\@tempa{#3}
	\ifx\@tempa\@empty\relax \else \textit{#3} \fi \par
}
